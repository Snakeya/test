=char ch = 253;
ch  = ch + 7;

//253->1111 1101->int类型，补符号位->11111111 11111111 11111111 11111101
			               +00000000 00000000 00000000 00000111（7）
    			           = 1 00000000 00000000 00000000 00000100
			           ->char 高位舍弃->00000100    
					            

printf("%d\n",ch);//把ch当作有符号输出->00000100->00000000 00000000 00000000 00000100->4						
printf("%u\n",ch);//把ch当作无符号输出->00000100->00000000 00000000 00000000 00000100->4
printf("%c\n",ch);//把ch以字符形式输出->4->EOT
printf("%x\n",ch);//把ch以16进制输出->4->4

unsigned ch = 253;

				     +00000000 00000000 00000000 00000111（7）
    				     =00000000 00000000 00000001 00000100
			                    ->char 高位舍弃->00000100
